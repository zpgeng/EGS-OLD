423 #define STARPU_ATOMIC_SOMETHING(name,expr) \
424 static __starpu_inline unsigned starpu_atomic_##name(unsigned *ptr, unsigned value) \
425 { \
426  unsigned old, next; \
427  while (1) \
428  { \
429  old = *ptr; \
430  next = expr; \
431  if (_starpu_cmpxchg(ptr, old, next) == old) \
432  break; \
433  }; \
434  return expr; \
435 }
