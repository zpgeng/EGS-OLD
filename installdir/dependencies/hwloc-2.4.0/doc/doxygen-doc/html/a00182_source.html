415 static __hwloc_inline int
416 hwloc_plugin_check_namespace(const char *pluginname __hwloc_attribute_unused, const char *symbol __hwloc_attribute_unused)
417 {
418 #ifdef HWLOC_INSIDE_PLUGIN
419  void *sym;
420 #ifdef HWLOC_HAVE_LTDL
421  lt_dlhandle handle = lt_dlopen(NULL);
422 #else
423  void *handle = dlopen(NULL, RTLD_NOW|RTLD_LOCAL);
424 #endif
425  if (!handle)
426  /* cannot check, assume things will work */
427  return 0;
428 #ifdef HWLOC_HAVE_LTDL
429  sym = lt_dlsym(handle, symbol);
430  lt_dlclose(handle);
431 #else
432  sym = dlsym(handle, symbol);
433  dlclose(handle);
434 #endif
435  if (!sym) {
436  static int verboseenv_checked = 0;
437  static int verboseenv_value = 0;
438  if (!verboseenv_checked) {
439  const char *verboseenv = getenv("HWLOC_PLUGINS_VERBOSE");
440  verboseenv_value = verboseenv ? atoi(verboseenv) : 0;
441  verboseenv_checked = 1;
442  }
443  if (verboseenv_value)
444  fprintf(stderr, "Plugin `%s' disabling itself because it cannot find the `%s' core symbol.\n",
445  pluginname, symbol);
446  return -1;
447  }
448 #endif /* HWLOC_INSIDE_PLUGIN */
449  return 0;
450 }

465 static __hwloc_inline int
466 hwloc_filter_check_pcidev_subtype_important(unsigned classid)
467 {
468  unsigned baseclass = classid >> 8;
469  return (baseclass == 0x03 /* PCI_BASE_CLASS_DISPLAY */
470  || baseclass == 0x02 /* PCI_BASE_CLASS_NETWORK */
471  || baseclass == 0x01 /* PCI_BASE_CLASS_STORAGE */
472  || baseclass == 0x0b /* PCI_BASE_CLASS_PROCESSOR */
473  || classid == 0x0c04 /* PCI_CLASS_SERIAL_FIBER */
474  || classid == 0x0c06 /* PCI_CLASS_SERIAL_INFINIBAND */
475  || baseclass == 0x12 /* Processing Accelerators */);
476 }

494 static __hwloc_inline int
495 hwloc_filter_check_keep_object_type(hwloc_topology_t topology, hwloc_obj_type_t type)
496 {
497  enum hwloc_type_filter_e filter = HWLOC_TYPE_FILTER_KEEP_NONE;
498  hwloc_topology_get_type_filter(topology, type, &filter);
499  assert(filter != HWLOC_TYPE_FILTER_KEEP_IMPORTANT); /* IMPORTANT only used for I/O */
500  return filter == HWLOC_TYPE_FILTER_KEEP_NONE ? 0 : 1;

507 static __hwloc_inline int
508 hwloc_filter_check_keep_object(hwloc_topology_t topology, hwloc_obj_t obj)
509 {
510  hwloc_obj_type_t type = obj->type;
511  enum hwloc_type_filter_e filter = HWLOC_TYPE_FILTER_KEEP_NONE;
512  hwloc_topology_get_type_filter(topology, type, &filter);
513  if (filter == HWLOC_TYPE_FILTER_KEEP_NONE)
514  return 0;
515  if (filter == HWLOC_TYPE_FILTER_KEEP_IMPORTANT) {
516  if (type == HWLOC_OBJ_PCI_DEVICE)
517  return hwloc_filter_check_pcidev_subtype_important(obj->attr->pcidev.class_id);
518  if (type == HWLOC_OBJ_OS_DEVICE)
519  return hwloc_filter_check_osdev_subtype_important(obj->attr->osdev.type);
520  }
521  return 1;
522 }
